#include <P16F877.INC>
    SAYI1   EQU 0X20
    SAYI2   EQU	0X21
    SAYAC1  EQU	0X22
    SAYAC2  EQU	0X23
    GECIK1  EQU	0X24
    ORG	0X000
    GOTO    ANA_PROGRAM
ANA_PROGRAM
    BANKSEL TRISB
    BSF	TRISB,0
    BSF	TRISB,1
    BSF	TRISB,2
    MOVLW   0X00
    MOVWF   TRISD
    BANKSEL PORTB
    CLRF    PORTD
    CLRF    PORTB
SAYI1_ARTIR
    BTFSS   PORTB,0
    
SAYI2_ARTIR
    BTFSC   PORTB,1
    INCF    SAYI2
    GOTO    SAYI1_ARTIR
SAYI_TOP
    BTFSC   PORTB,2
    GOTO    TOP

TOP
    MOVFW   SAYI1
    ADDWF   SAYI2,0
    MOVWF   PORTD
    
    END
