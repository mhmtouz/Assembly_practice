    LIST P=16F877A
    #INCLUDE <P16F877A.INC>
    __CONFIG H'3F31'
    ORG 0X000
    GOTO BASLA
BASLA
    MOVLW 0X20
    MOVWF FSR
    
DONGU
    MOVLW 0X01
    MOVWF INDF
    
    INCF FSR,F
    MOVLW 0X40
    SUBWF FSR,W
    
    BTFSS STATUS,Z
    GOTO DONGU
    GOTO    ISLEM
ISLEM
    MOVLW 0X02
    MOVWF INDF
    
    DECF FSR,F
    MOVLW 0X20
    SUBWF FSR,W
    
    BTFSS STATUS,Z
    GOTO ISLEM
    END
